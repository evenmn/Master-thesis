\newcommand{\mframe}[3]{
	\frame{
		% PROGRESSION WHEEL 
		\begin{textblock}{3}(0.5,8)
			\begin{tikzpicture}[x=1mm,y=1mm]
			\foreach \x in {1,2,...,\inserttotalframenumber} 
			\draw [gray!30,very thick] (\x/\inserttotalframenumber*360:3) -- (\x/\inserttotalframenumber*360:4);
			\foreach \x in {1,2,...,\insertframenumber}
			\draw[color0,very thick] (\x/\inserttotalframenumber*360:3) -- (\x/\inserttotalframenumber*360:4);
			\node at (0,0) [] {\tiny{\textbf{\textcolor{black}{\insertframenumber}}}};
			\end{tikzpicture}
		\end{textblock}
	
		% UIO LOGO
		\begin{textblock}{3}(4.2,8.3)
			\centering
			\includegraphics[scale=.2]{../Images/UiO_Seal_A_ENG_cmyk.eps}
		\end{textblock}
		\frametitle{\textcolor{color0}{#1}}
		\framesubtitle{#2}
		\vspace{-0.5cm}
		\scriptsize{#3}}
}
