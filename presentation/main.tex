\mframe{\frametitle{Harmonic oscillator}

\begin{equation}
\hat{\mathcal{H}} = \sum_{i=1}^{N} \bigg(-\frac{1}{2} \nabla_i^2 + \frac{1}{2} \omega^2r_i ^2\bigg)

\mframe{\frametitle{Quantum dots}
	
	Many-body problem
	
	\begin{equation}
	\hat{\mathcal{H}} = \sum_{i=1}^{N} \Big(-\frac{1}{2} \nabla_i^2 + \frac{1}{2} \omega^2r_i ^2\Big) + \sum_{i<j} \frac{1}{r_{ij}}
	\end{equation}
	
}

&= \int d\bs{r} E_L(\bs{r})P(\bs{r})\\
& \approx \frac{1}{M}\sum_{i=1}^ME_L(\bs{r}_i)

\end{equation}
